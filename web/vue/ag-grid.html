获取sort ​

ts
  const sortChanged = (e: AgGridRowEvent) => {
    let arrs = e.api.getAllGridColumns();
    let hasSort = arrs?.some(x => {
      if (x.getSort()) {
        colum = {
          row: x,
          sort: (x.getSort() as 'asc' | 'desc')
        };
        return x;
      }
    });
    if (hasSort) emit('_sortChanged', colum)
    else emit('_sortChanged', null)
  }
