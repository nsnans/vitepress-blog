在vitepress使用 ​

我的:

.vitepress文件夹下，新建theme文件夹，里面建index.ts文件，代码:

ts
import DefaultTheme from 'vitepress/theme-without-fonts'
import layout from '../../src/layout'
import type { Theme } from "vitepress";

const t: Theme = {
    extends: DefaultTheme,
    Layout: layout
}
export default t

layout.tsx

tsx
import DefaultTheme from 'vitepress/theme';
import { defineComponent, h } from 'vue';
import Utteranc from './Utteranc';
import { unref, computed } from 'vue'
import { useData } from 'vitepress'


export default defineComponent({
    setup() {
        const { Layout } = DefaultTheme
        const { page } = useData()
        const isHome = computed(() => unref(page)?.filePath === 'index.md')

        return () => {
            return <Layout v-slots={{
                'doc-after': isHome.value ? undefined : () => h(<Utteranc />,
                    {
                        style: { marginTop: '20px' },
                        key: page.value.filePath,
                    }
                )
            }} />
        }
    }
})

Utteranc.tsx

tsx
import { defineComponent, onMounted, ref, type Ref } from 'vue';


export default defineComponent({
    setup() {
        const commentRef: Ref<HTMLElement | null> = ref(null);
        onMounted(() => {
            try {
                if (commentRef.value) {
                    const script = document.createElement('script');
                    script.src = 'https://utteranc.es/client.js';
                    script.async = true;
                    script.crossOrigin = 'anonymous';
                    script.setAttribute('repo', 'XXX/comments');
                    script.setAttribute('issue-term', 'pathname');
                    script.setAttribute('label', 'Comment');
                    script.setAttribute('theme', 'preferred-color-scheme');
                    commentRef.value.appendChild(script);
                } else {
                    console.warn(`Comments won't load because the commentRef element is null.`);
                }
            } catch (error) {
                console.error('Comments loading failed.', error);
            }
        })

        return () => <div ref={commentRef} id="comment"></div>
    }
})

WARNING

代码很简单，改成vue风格的代码就可以
