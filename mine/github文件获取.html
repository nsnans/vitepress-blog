通过HTTP获取GitHub私有仓库中的文件夹内容，并将这些文件写入本地指定目录 ​

ts
const fs = require('fs');
const path = require('path');


const owner = 'nsnans';//GitHub用户名
const repo = 'vitepress-blog-md';//私有仓库名
const folderPath = 'mine';//文件夹路径
const token = atob('Z2hwX2xUTldQRkQwVGY0UTg3YjJFVGpQQnFpZUFxY3h3WjFmd3dPTg==') ;//个人访问令牌
const localDirectory = 'mds';//本地指定目录

const fetchFolderContentsFromGitHub = async (owner: any, repo: any, path: any, token: any) => {
    const url = `https://api.github.com/repos/${owner}/${repo}/contents/${path}`;

    const response = await fetch(url, {
        headers: {
            'Authorization': `token ${token}`,
            'Accept': 'application/vnd.github.v3.raw'
        }
    });

    if (!response.ok) {
        throw new Error(`获取文件夹内容失败：${response.statusText}`);
    }

    const folderContents = await response.json();
    return folderContents;
};

const fetchFileContent = async (fileUrl: string | URL | Request, token: any) => {
    const response = await fetch(fileUrl, {
        headers: {
            'Authorization': `token ${token}`,
            'Accept': 'application/vnd.github.v3.raw'
        }
    });

    if (!response.ok) {
        throw new Error(`获取文件内容失败：${response.statusText}`);
    }

    const fileContent = await response.text();
    return fileContent;
};

const writeFileToLocal = (filePath: any, content: string | NodeJS.ArrayBufferView) => {
    const dir = path.dirname(filePath);
    if (!fs.existsSync(dir)) {
        fs.mkdirSync(dir, { recursive: true });
    }
    fs.writeFileSync(filePath, content);
};

const downloadFolderFromGitHub = async (owner: string, repo: string, folderPath: string, token: string, localDirectory: string) => {
    const folderContents = await fetchFolderContentsFromGitHub(owner, repo, folderPath, token);

    for (const item of folderContents) {
        if (item.type === 'file') {
            const fileContent = await fetchFileContent(item.download_url, token);
            const localFilePath = path.join(localDirectory, item.path);
            writeFileToLocal(localFilePath, fileContent);
        }
    }
};

// 使用示例
downloadFolderFromGitHub(owner, repo, folderPath, token, localDirectory)
    .then(() => {
        console.log('文件夹下载并保存到本地成功');
    })
    .catch(error => {
        console.error(error);
    });
